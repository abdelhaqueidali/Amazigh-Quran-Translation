68
تازّيت (أغانيب)
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، 1 س ييسم ن ڕبّي أمالّاي أمسمولّو، **نّون**: ڕبّي كا أد ئسّن ماد سرس ئواتّس، ئݣّولّا ڕبّي س **نون** د تازّيت (أغانيب): **لقالام**، د كرا سرس تّاران
2 أمكي تݣيت أ أمازان أنافال، س تنافوت لّي أك ئفكا ڕبّي نّك
3 هانّ لّانت أك تغراد ؤر تّوباينين
4 هانّ كيّي لّانت فلّاك تيميتار فولكينين
5 هانّ راد تژڕت كيّي، غيكلّي راد ژڕين نتّني
6 مان وانّا ݣيݣون ئݣان أنافال
7 هانّ ڕبّي نّك ئسّن غوالّي ئفرغن ف ؤغاراس نّس، غيكلّي ئسّن ماد ديس ئمونّ
8 هانّ أد ؤر تسݣدت ئ ئمسّكيركّسن
9 هان ران ݣيك أد فلّاسن تسيفسيست، أيوان ؤلا نتّني سيفسيسن فلّاك، س توضّڕت كيّي ؤلا نتّني
10 هانّ أد ؤر تسݣدت ئ كرا ئݣان ئمݣيلّي ف ماد ئݣان ئنزݣي ؤلا ماد ئݣان أبنّان، سول ئݣ أمتّولكو، أشكو ؤر ييلي أتيݣ
11 هانّ ئݣا أقجّام: أر يادّرا ميدّن س ݣار، غيكلّي تاباليست ݣر ميدّن
12 ئݣ أمتّوݣي ن وافولكي، سول ئمون ݣيس كرا ئݣان ݣار
13 ئݣا زون د أفوناس، غ كرايݣات تاسݣا، سول د ؤݣّار ن غيكانّ أحّرام، ؤر ئتّوسّان باباس؟
14 ئفكا أس ڕبّي أيادا د تاروا، سول ياݣْࢫي **ليسلام**؟
15 ئنّا ڕبّي: ئغ أد فلّاس تّوغرين أيلان نّغ، ئنّا: غاياد ئݣا أ غار ؤمين ن ئمزوُورا
16 ئنّا ڕبّي: راد أس نسكر ݣار تاماتارت غ ئݣّي ن ؤغنزور نّس ما س ئتّوسّين
17 ئنّا ڕبّي هانّ نمرّت أيت **ماكّا**، غيكلّي نمرّت أيت تورتيت، أيلّيغ ݣّولّان أ سار ت مݣرن غ تييْلݣي
18 ماشّ ؤر نّين ئغ ئرا ڕبّي
19 ئسّوتل أس يان ؤمسّوتل غ دار ڕبّي نّك طّاصن نيت سول، زعما ئسغر ت ڕبّي، ئسنݣيري تن د تييّافوت نّس
20 س تݣا زون د أساغور ئسݣّانّ
21 ئس ؤكان ئفاو ؤزال، أر ياقّرا كرا ݣيس ئ كرا
22 أر تّينين نكرات زيكّ س تمݣرا نّون ئغ راد تمݣرم؟
23 هانّ دّان س تبحيرت ماشّ س تونتلا
24 نّان: هانّ أد فلّاون ؤر ئكشيم غ واسّ أد غ تبحيرت كرا ن ؤمژلوض
25 نكرن تايلݣي س تازّلا، نّان نژضاڕ أد تّ (أ ستّ) نمݣر
26 ئس تّ ؤكان ژڕان تجدر نّان: هانّ نضّڕ أغاراس سرس
27 نغ د ؤلا نكّني نݣا ئمژلاض؟
28 ئنّا غوالّي ݣيسن ئسوِينݣيمن: ماخّ ؤر أون نّيغ أيوا، تّرات ڕبّي أد أون ئسّورف؟
29 أيوان نّان: تومايت ئ ڕبّي نّغ، هانّ نكّني نكّا تّ ئنّ نݣا ئنژضامن
30 يايوا كرا ݣيسن أر ئتّمناد غ كرا، أر تّمزاوارن
31 نّان: أحّ ن ئغنان نّغ، نكّني نكّا تّ ئنّ نزري أوتّو ݣراغ د ڕبّي
32 بار ڕبّي نّغ أد أغ ت ئسنفل س ماد تّ (ما ستّ) يوفن، هانّ نكّني نژوژض أ غار غ ڕبّي نّغ
33 غيكانّ د تمّارا، ماشّ تامّارا ن تمݣّاروت تݣّوت، متاد ئس ݣان ويد سّنّين
34 هانّ غوِيلّي كصوضنين ڕبّي، لّانت أسن غ دار ڕبّي نّسن تورتيتين ن تنافوت: **جانّات نّاعيم**
35 ماخّ ئس راد نسݣيدّي **ئموسلمن** د ئكنّافن؟
36 ما كْࢫن ياغن، مانيك س تغالم؟
37 نغ د ئس دارون كرا ن ؤدليس أر ݣيس تاقّرام؟
38 ئلّا أون ݣيس ماد تستّيم
39 نغ د ئس دارون ئمقّونّ تّياران فلّاغ أر أسّ ن ؤسّحسو؟ فاد أد تسكرم ماد ترام
40 أيوا، ساقساتن مان وانّا س ؤمژن؟
41 نغ د ئس دارسن ئمدراو؟ أوا، أد د أوين ئمدراو نّسن ئغ ݣان ئموغْࢫژانّ؟
42 كتي د أسّ لّي غ راد بيدّن ف يان ؤضاڕ، فاد أد حسون س أسن ئتّونّا أيوا، ݣّونتويات ماشّ ؤر أس ژضاڕن
43 أيوان سغلولّين ألّن نّسن، أر تن تسخنونوس تنّوݣرت، هانّ كّان تّ ئنّ أر أسن ئتّونّي ياد أيوا ݣونتويات أيلّيغ ݣان ئمدواس
44 أجّ ئي س غوالّي ئسّكيركّسن **لقْࢫڕان** أد، أد ت مرّتغ، هانّ راد أسن نسالو، ئميكّ س ئميكّ سول ؤر أس فريكن، أر ديغ تْتّياماژن
45 سول راد تن سݣروغ، أشكو تاكنداوت ئنو تدوس
46 ماخّ ئس أد تن تساقسات غ تغراد؟ سول نتّني ؤسين أزاژو (أژاژو*؟) ن تفݣورت؟
47 نغ د ئس دارسن أدليس ئنتلن أر ݣيس أقّران؟
48 أيوا، ژيضڕ ئ ئزرف ن ڕبّي نّك، هانّ أر ؤر تݣت زون د بو ؤسلم، أيلّيغ ئغرا ئ ڕبّي ماشّ ئݣا أمغيلف
49 متاد ئس ت ئد ؤر تنّيل تملّاوت زغ دار ڕبّي نّس، أ سار نّ ئضڕ غ تاغارت سول ئݣ مژّي
50 ماشّ ئستيت ڕبّي نّس يايوا ئسكر ݣيس يان غ غوِيلّي روانين
51 ها نّ علاين أكّْࢫ جدرن غوِيلّي نبلنين س ئژلّوݣن نّسن، أيلّيغ سّفلدن ئ **لقْࢫڕان** أيوان أر تّينين: هانّ ئݣا أنافال
52 نتّا ؤر ئݣي بلا أسّكتي ئ كرا ئلّان
