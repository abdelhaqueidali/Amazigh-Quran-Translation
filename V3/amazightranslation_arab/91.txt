91
تافوكت
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي أد ݣّالغ س تافوكت د ؤزنژاڕ نّس
2 د وايّور ئغ ت ئد ئطّفاڕ
3 د واسّ ئغ ت ئد ئسّوݣّا
4 د ييض ئغ ت ئدل
5 د ئݣنّا د ما تّ ئسّكين
6 د واكال، د ما تّ ئفسمن
7 د ئمان، د ما تّ ئساستوان
8 ئمل أس أغاراس ئفرغن د واد يوغدن
9 أر ئجّنجم ڕبّي واد ئسغوسن ئخف نّس
10 غيك لّي ئسّوكل غوالّي ت ئنخّلن
11 ها نّ تسّكير ݣيس **تامود** أمازان نّسن، س ؤژضام نّسن
12 ئلّيغ أيتّازّال واشّ نّسن، ئسكر غايلّي خف د مساشكا؟
13 ئنّا أسن ؤمازان ن ڕبّي: أد ت ئݣان د **صالح** ها نّ أد ؤر تسليم تاڕامت ن ڕبّي د تيسّي نّس ݣ واسّ نّس؟
14 ماشّ سّكيركّسن ت، أيوْڕن نغين تاڕامت، غيك أنّ أ خف تن ئسرفوفنڕبّي نّسن س ؤبكّاض نّسن، يايوا ئسݣون فلّاسن تامّارا، ؤر ݣيسن ئنجم مقّار د يان
15 أشكو ڕبّي ؤر ئكصوض، ما راد د ئݣّرو ئ غيك أنّ؟ نغ د غوا لّي ئنغان تاڕامت، ؤر نّ ئتام ما راد ئݣّرو؟
