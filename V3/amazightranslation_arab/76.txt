76
أفݣان
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، هانّ ئزري ياد كرا ن ؤزمز ن وانّاز ف ؤفݣان لّي س تّ ئنّ ئكّان ؤر ئلي أتيݣ
2 ئنّا ڕبّي: هانّ نكّني نسكر أفݣان غ ؤبرغويْ ئرعين، فاد أد ت نارم، نايوا نسكر أس أسّفلد د ئژڕي
3 هانّ نملا أس أبريد، ف أد ئݣ أنافلاس، نغ أمنابال؟
4 هانّ نسمّوتݣ ئ ئمنابالن ئسكراف، د تايقارين د تمژلا
5 هانّ غوِيلّي روانين أر سّان غ يان ؤفشكو تيسّي ئمّيمن
6 أد ت ئݣان د يان ؤغبالو أر ݣيس سّان ئسمݣان ن ڕبّي، أيوان سّنݣين ت س أينّا ران
7 غوِينّ أر جّنجومن أينّا س ؤسيِين غيكلّي تّيكصوضن، يان واسّ لّي مو ئنݣي ؤژوض نّس
8 د أر أكّان تيرمت ئ ؤمژلوض واخّا ت زوژضن (ژوژضن)، غيكانّ د ئݣيݣيل، د واد ئتّياماژن
9 سول أر تّينين: هانّ أر أون ناكّا تيرمت ف ؤدم ن ڕبّي، هانّ ؤر ݣيݣون نري تيغراد، ؤلا تولغيتين
10 هانّ نكّني، نكصوض أ غار ڕبّي نّغ غ يان واسّ ئكرسن يوسّن
11 غيكانّ أ ف تن يارا ڕبّي ف ؤژوض ن واسّ أنّ، ئفك أسن أسمرقا ن ؤدم د تنومّرت
12 غيكلّي أسن ئفكا تيغراد نّسن أيلّيغ زّيضڕن، أد تنت ئݣان، د ؤكشّوم س تورتيت، سول ئݣ ؤمطّالسو تيملسيت نّسن
13 أيوان ژّالن ݣيس ف ئسوما، ؤر أد ݣيسن قروكن ئ تافوكت، ؤلا أسمّيض (أصمّيض)
14 سول أݣْࢫلن د فلّاسن ئسكّيضن ن تورتيت أنّ، ئكنو د فلّاسن ؤژواي نّس، سول س غيكلّي ت ران
15 أر أسن تّسوتولن وازّانّ س ئفشكا ن تّقّورت د ئزعكان سميرقينين
16 ئفشكان أنّ لّي سميرقينين، تّوسكارن غ نقّرت (ؤژڕف) أر تّاسين ؤنشك لّي سّان
17 أر أسن سّوان ئنژيڕ ئرعين د توجّوت
18 أد ت ئݣان د يان ؤغبالو ݣيس، ئتّوسّان س تاميمت
19 أر أسن تسّوتولن وازّانّ ݣانين ئمغلال، ئغ تن تانّيت تغالت تن ئس ݣان أمغوان ئنّا زوان
20 ئغ تانّيت غ غينّ، هانّ أر تّمنادت تانافيت، د تكلديت تاباراوت
21 لسان تيملسيت ن ؤمطّالسو أزيݣزاو (أزݣزاو)، د ؤسمرغاي، غيكلّي ؤݣلن تانّالين ن ؤژڕف (نّقْࢫرت)، سول أر أسن ئسّوا ڕبّي نّسن س أنژيڕ ئزدّيݣن
22 ئنّا أسن ڕبّي: هانّ غواد ئݣا أون تيغراد، أشكو أسكّير نّون ئفولكي
23 هانّ نكّني نزّوݣز نّ فلّاك **لقْࢫڕان** ئميكّ س ئميكّ
24 أوا، ژيضڕ ئ ؤزرف ن ڕبّي نّك، هانّ أد ؤر تسݣدت ئ ؤبكّاض ݣيسن ؤلا أمنابال
25 أيوا، بدر ئسم ن ڕبّي نّك غ تايلݣي د تادوݣات
26  تايوات تݣونتوت (تݣْࢫنتوت*؟) أس غ ييض، تمّويْت أس غ ييض ئغزّيفن
27 هانّ غوِيد أر تّيرين تامزرويْت، أيوان تّون يان واسّ ئژّاين لّي تن ئد ئقّانّ
28 نكّني نسكر تن، نسّيدوس أسكّير نّسن، ئغ نرا راد تن نسنفل س ماد تن ئرواسن
29 هانّ ختاد تݣا أسّوكتي (أسّكْࢫتي) ئ وانّا ئران أد ئمون د ئغاراس ن ڕبّي نّس
30 هانّ ؤر أد تّيريم بلا ئغ ئرا ڕبّي، هانّ ڕبّي ئݣا أمسّان أموسناو
31 أر ئسّكشوم وانّا ئرا س تملّاوت نّس، زعما تورتيت نّس، غيكلّي ئسمّوتݣ ئ ئنژضامن تامّار تاماݣزاژت (تاماݣژاژت)
