تاغرغارت-3
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، 1 س ييسم ن ڕبّي أمالّاي أمسمولّو، تاغرغارت لّي د ئقّانّ، لّيغ راد ياݣّْࢫ كرا غ سكرن تيكركاس
2 أحّ مانيك تݣا تغرغارت أنّ ئغ د توݣّا!؟ هانّ ئسرفوفن ن ڕبّي كرا تّ ئسّكيركسن
3 مان وانّا راد أك ئمل تاغرغارت أنّ لّي د ئقّانّ؟
4 هانّ **تامود**د **عاد** سّكيركّسن تاغرغارت نّس راد تيلي
5 أمّا **تامود** هانّ رفوفونّ س تغويّيت
6 أمّا **عاد** بانّ رفوفونّ س كيݣان ن واضو ئسغويّين
7 ئݣر ت فلّاسن ڕبّي سا واضان، د تام ؤسّان زدينين، س أر تّانّيت ميدّن ضڕان ݣيس زون د ئݣجوان ن ئݣوجّاف مّارنين
8 أوا، ئس أسن جّو تژليت أݣالوز؟
9 يوشكا د **فيرعاون** د ماد ت ئزوارن، د أيت ئغرمان س ؤبكّاض
10 أيوان مّزرين د ؤمازان ن ڕبّي نّسن، غيكانّ أ خ تن يومژ يان وامّاژ يوسّن
11 هانّ أيلّيغ أتّويْن وامان، نوسي كن (كْࢫن) ف ئݣي (ئݣّي) ن تاناوت
12 فاد أد أون ݣيس نسكر أسّكتي، أد ت حسون ئمزݣان (ئمزّاغ) سّفليدنين
13 هانّ ئغ ئتّوصواض غ ئسك يان ؤصواض
14 سيغ ئتّياساي واكال د ئدرارن نميقّيرن ݣراتسن، أيوان قرسّين س يات تكّليت
15 هانّ أكود أنّ (كودانّ) تلكم تغرغارت
16 س ئستغ ئݣنّا أشكو أكود أنّ (كودانّ) ئلمض
17 س بدّن (بيدّن) ئنيرن غ تسݣّيوِين (تسيݣيوِين) ن ئݣنوان، أيوان أسن تّام ݣيسن أدابو ن ڕبّي نّك غ واسّ أنّ (غاسّ أنّ) ف ئݣّي (ئݣي) ن ؤمݣڕاض (ؤمݣراض) نّسن
18 غ واسّ أنّ (غاسّ أنّ) أ راد تݣيم دّرست، فاد أد تحسوم، ؤر أد ݣيݣون ئنتل مقّار د ؤمويا (أميا) غ غايلّي تسكرم
19 ئنّا ڕبّي: هانّ وانّا مو ئتّوفكا ؤدليس نّس غ ؤفوس نّس أفاسي أر ئتّيني: أيوا أمژات غرات أدليس ئنو ئفولكين
20 هانّ نكّي كّيغ تّ ئنّ غ تودرت ئنو، سّنغ ئس راد تّوسقسيغ
21 ئنّا ڕبّي: غواد، هانّ ئلّا غ وادّاد ئفجّيجن
22 غ تورتيت ياتّويْن
23 أژواي ن تژڕمين (تژرمين) نّس ياژ د أفوس بلا تاكرّايت
24 أر أسن ئتّياونّا (ئتّونّي): أيوا أر شتّايات أر تسّام، هانّ ؤرغود أون س غايلّي تزّريم غ ؤسّان زرينين غ تودرت
25 أمّا وانّا مو ئتّوفكا ؤدليس نّس غ ؤفوس نّس أژلماض، هانّ أر ئتّيني: مرا هلّي ؤر ئيي ئتّوفكا (ئتّوفكي) ؤدليس ئنو؟
26 ؤلا سّنغ ما غ راد تّوسقسيغ
27 مرا هلّي تݣا تمتّانت (**لموت**) أوتّو ݣر ئيي د تودرت أد؟
28 أشكو ؤر فلّا يوسيْ وايادا ئنو مقّار د أميا
29 غيكلّي ئي تفل تدوسي ئنو
30 كودنّا أد ئتّونّا: أيوا، أمژات ت، كرفات ت
31 تاويم ت تݣرم (تلوحم) ت ئنّ غ ؤژيژن
32 أيوا، ڕژييات (رژييات) ت غ سّفود ن ؤزّال، تيغزي نّس، ئلّا ݣيس سا ئد مراو (سامراو نيغ د كڕاض واݣنارن د مراو) ن ئغيل
33 هانّ ئكّا تّ ئنّ ؤر ئفليس س ڕبّي مقّورن
34 غيكلّي ؤر أد ئكّات تاكرّايت ف ؤمتّشو ن ئمژلاض
35 هانّ غ واسّ أد (غاسّ أد) ؤر دارس غ غيد أمدّاكّل
36 ؤلا تيرمت بلا غ ؤڕصض (ؤرصض) لّي ئنݣين غ أيت ؤرواسّ
37 ؤر أد ت ئشتّا بلا ئمنابالن
38 ئنّا ڕبّي: أد ݣّالغ س ماد تّمنادم
39 ؤلا ماد ؤر تّمنادم
40 أمكي ئݣا **لقْࢫڕان** أد بلا تينّيت ن ؤمازان ئڕوان (ئروان)
41 هانّ ؤر ئݣي ؤرارن ن ؤمارير، ماشّ ئدروس أد تفلسم
42 ؤلا ئݣا أوال ن ؤݣزّان، ماشّ ئدروس أد تكتّيم
43 هانّ أد ئݣا د ؤݣوز زغ دار ڕبّي ن كرا ئلّان
44 هانّ متاد ئس فلّاغ ئسكر كرا ن تكركاس
45 أ سار ت نسرفوفن س تدوسي نّغ
46 سول نبّي أس أژاڕ (أژار)ن ؤل
47 مقّار د يان ݣيݣون ؤر ئژضاڕ (ئژضار) أد فلّاس ئرار
48 هانّ **لقْࢫڕان** ئݣا أسّكتي ن غوِيلّي كصوضنين ڕبّي
49 هانّ نكّني نّسن ئس ݣيݣون لّان ئمسّكيركّسن
50 هانّ ئݣا تامݣڕاژت (تامݣراژت) ئ ئمنابالن
51 هانّ ئݣا ئنزݣي غ ؤر تلّي توردا
52 أيوا، مّويْ س ييسم ن ڕبّي نّك أمقّران
