64
تاغندافت
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، أر ئتّومّويْ ئ ڕبّي كرا ئلّان غ ئݣنوان، د كرا ئلّان غ واكال، تاݣلديت تݣا تينّس، غيكلّي تݣا تامايت تينّس، نتّا ئژضاڕ ئ كرا ئلّان
2 نتّا أد كْࢫن ئسكرن، ݣيݣون أمنابال، غيكلّي أنافلاس، هانّ أكوش ئݣا أمانّاي ن كرا تسكارم
3 ئسكر ئݣنوان د واكال س ؤنزݣي، يايوا ئسّولف كْࢫن، سول ئسّفالكي ئولافن نّون، هانّ دارس أد نّ ترا توغولت
4 ئسّن ماد ئلّان غ ييݣنوان د واكال، غيكلّي ئسّن ماد ت نّحضام، ؤلا ماد د تسّوݣّام، هانّ ڕبّي ئݣا أماسّان ن كرا تسوِينݣيمّ
5 ئس كن ئد ؤر يوشكي ؤنغميس ن غوِيلّي نبلنين غ ماد ئزرين؟ ها نّ مضين تاسّاست ن وارّازن نّسن، غيكلّي أسن تلّا تمّارا تاماݣزاژت
6 غيكانّ أشكو كّان تّ ئنّ ئمازانّ نّسن أر أسن د تّاوين تيميتار، نّان: ماخّ ئس د أفݣان أ راد أغ ئسمون د ؤغاراس؟ أيوان نبلن أنفن ئ ؤغاراس ن ڕبّي، أوا ها نّ ڕبّي ؤر فلّاسن ئلكيم، أشكو ڕبّي ئݣا أمادواس أمتّوݣاضاي
7 غالن غوِيلّي ئنبلنين، ئس ؤر سار راد د نكرن غ ئسمضال نّسن، ئني: عاد، تاݣالّيت س ڕبّي أ س راد سول تنكرم، سول تْتّوسقسيم غ كرا تسكرم، ها نّ ئݣا أموسّن ن كرا تسكارم
8 أيوا، فلسات س ڕبّي د ؤمازان نّس، د تيفاوت لّي د نسّوݣز: **لقوران** ها نّ أكوش ئݣا أموسّن ن كرا تسكارم
9 كْࢫتي د أسّ لّي غ راد كْࢫن ئسمون ڕبّي غ واسّ ن تغرغارت ؤ، أد ت ئݣان د واسّ ن ؤنموقّار، غوانّ أد ئݣان أسّ ن تغندافت، ها نّ وانّا ئفلسن س ڕبّي، أر ئسكار ماد ئفولكين راد فلّاس نسّيتّي ئبكّاضن نّس، نايوا نسّكشم ت س تورتيتين لّي مو نݣين ئسافّن غ دّاوات نّسنت، أيوان زدغن بدّا ݣيسنت، غيك أنّ أد ئݣان أمركيد أنمغور
10 أمّا ويلّي نبلنين سّكيركّسن تيميتار نّغ، ها نّ وينّ أد ئݣان أيت لعفيت راد بدّا ݣيسزّدغن، ئخّ غ توغولت!
11 ها نّ ؤر أد تكّات كرا ن تواغيت بلا س تورايت ن ڕبّي، وانّا ئفلسن س ڕبّي، ها نّ راد ئسمون ؤل نّس د ؤغاراس، أشكو أكوش ئݣا أماسّان ن كرا ئلّان
12 أيوا سݣدات ئ ڕبّي، تسݣدم ئ ؤمازان، ئغ توݣيم، ها نّ أد أكّْࢫ ئلّان خف ؤمازان نّغ، د ؤسّلكم ئفاون
13 يايوا ڕبّي ئملا أغ: ها نّ **لّاه** {عازّا واجّالّا} ؤر ئلّي كرا ن ڕبّي بلا نتّا، ها نّ أكوش أ مو تّݣّان ئنافلاسن تاوريت نّسن غ ؤفوس نّس
14 وا ويلّي فلسنين: ها نّ أماݣزار نّون، ئلّا غ تمغارين نّون د تاروا نّون، أوا ئكصوضات ݣيسن، ماشّ ئغ تسيفسيسم، تايوام تزّريم أسن ت، تسّورفم أسن، ها نّ أكوش ئݣا أمسّاروف أمسمولّو
15 ها نّ أد أكّْࢫ ئݣا وايادا نّون د وارّاو نّون د تمروايت، ها نّ تيغراد ݣّوتنين لّانت غ دار ڕبّي
16 أيوا ئكصوضات ڕبّي س ؤنشك نّا مو تژضاڕم، تسّفلدم ئ ماد أون ئتّونّان، تايوام تسݣدم، سول أر تاكّام، ها نّ ئݣا أفولكي ئ ئخفاون نّون، ها نّ وانّا ئحضان ئخف نّس غ تاسّاست ن وايادا نّس، ها نّ وينّ أد ئݣان ئمركيدن
17 ئغ أد تڕطّلم ئ ڕبّي أڕطّال ئفولكين، ها نّ راد أون ت ئرار خف سنات تيكّال، سول ئسّورف أون، أشكو أكوش ئݣا أسنّمّار أنمّازّو
18 نتّا أد ئݣان أماسّان ن كرا ئحضان، ؤلا كرا د يوݣّان، غيكلّي ئݣا أندابو أموسناو
