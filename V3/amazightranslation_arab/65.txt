65
تانڕژومت
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، وا أمازان : ئغ راد تلّفم ئ تمغارين، أيوا لّفات أسنت غ ؤزمز ن تغوسي نّسنت، سول تسّوضنم ؤغوسن، أيوا ئكصوضات ڕبّي {جالّا جالالوهو} لّي ئݣان ڕبّي نّون، ها نّ أد تنت ؤر تسّوفوغم غ تݣْࢫمّا نّسنت، ؤلا تّوسّوفاغنت بلا ئغ سكرنت كرا ن توژبيلت ؤر ئنتيلن، ختّي نّ أد ئݣان ئوتّا ن ڕبّي، وانّا ئزرين ئوتّا ن ڕبّي ها نّ ئژضم ئخف نّس، ها نّ ؤر تسّينت أد د ؤر ئسّݣرو ڕبّي ضاڕات غيك أنّ ن تغاوسا ؟
2 ئغ سمدنت أزمز نّسنت، أوا رارات تنت س ؤفولكي نغ ديسنت تبضام س ؤفولكي، ماشّ سكرات ئناݣان غ أيت ؤسومو ݣيݣون، تفكيم توݣّا ف ڕبّي غيك أنّ نّون أد ئݣان أسلويُو ئ وانّا ئفلسن س ڕبّي، د واسّ أمݣّارو، وانّا ئكصوضن ڕبّي راد أس ئمّال ماني غ ئتّفّاغ
3 غيك لّي راد أس ياكّا غ ما نّ ؤر ئتام، وانّا ت ئݣان غ ؤفوس ن ڕبّي، ها نّ ئقّادّا ت، ها نّ ڕبّي راد ئسّلكم تاغاوسا نّس، أشكو ڕبّي ئسكر ياد ئ كرا ئلّان أزمز نّس
4 أمّا تيمغارين لّيغ تلّا تغونفي ؤر أسليلينت غ لوهلات نّون، ئغ دارون توردا، ها نّ ؤغوس نّسنت كڕاض ييرن، غيك أنّ د ختيلّي ؤر سلينين، أمّا تيماراوين ها نّ أزمز نّسنت أرد أرونت، وانّا ئكصوضن ڕبّي، راد فلّاس ئسّيفسيس كرا فلّاس يورا
5 غيك أنّ ئݣا أزرف ن ڕبّي، ئسّوݣز ت ئد فلّاون، وانّا ئكصوضن ڕبّي راد فلّاس ئسّينيف ئبكّاضن نّس، سول ئسّوݣت أس تيغراد
6 أيوا، سّزدغات تنت غ غيلّي غ تزدغم واخّا تاما نّون، ها نّ أد تنت ؤر ئسنّوكموم تفكيم أسنت تيكڕژي، ئغ ژّاينت س وارّاو، فكات أسنت أرد ديغ ؤرونت أرّاو نّسنت، ديغ أد أون سّوضوضن أرّاو نّون، فكات أسنت تيغراد نّسنت، تايوام تسّاشكيم د ݣراتون س ؤفولكي، ئغ فلّاس ؤر توشكيم، ئفك أرّاو ن تايّاض أد تّ ئسّوموم
7 ئقّان د أد ئفك بو وايادا تيغراد س ؤنشك لّي مو ئژضاڕ، ماشّ وانّا مو ئدروس وايادا نّس يايوا أر ياكّا غ غايلّي أس ئفكا، ها نّ راد تّ ئفسي ڕبّي ضاڕات ن ماد توسّ
8 ها نّ منشك غ أيت ئغرمان أد ينفن ئ ؤغاراس ن ڕبّي نّسن، د ئمازانّ نّس، ئنّا ڕبّي : ها نّ راد تن نسّحسو، أماكڕاژ سول نمرّت تن تامّارا يوسّن
9 أيوان، مضين تاسّاست ن وارّازن نّسن، غيك لّي تݣا تݣيرا ن تغاراست نّسن أبنّان
10 ها نّ ئسمّوتݣ أسن ڕبّي تامّارا تاماݣزاژت (تاماݣژاژت)، أيوا ئكصوضات ڕبّي أ أيت ؤنلّي لّي فلسنين، ها نّ ئسّوݣز د فلّاون ڕبّي **مقوران**
11 غيك لّي د يوزن أمازان أر فلّاون ياقّرا تيميتار ن ڕبّي تيمسّفيوِين، فاد أد ئسّوفغ ويلّي فلسنين سكرن ماد ئفولكين غ تيلّاس كشمن س تيفاوت، ها نّ وانّا ئفلسن س ڕبّي، أر ئسكار ماد ئفولكين راد ت نسّكشم س تورتيتين لّي مو نݣين ئسافّن غ دّاوات نّسنت، أيوان زدغن بدّا ݣيسنت، وانّ ئسّوݣت أس ياد ڕبّي تيكّي
12 ڕبّي لّي ئسكرن سا ئݣنوان، ئسكر غ واكال زون د نّسن، أر تفتّو تايري ن ڕبّي ݣراتسن، فاد أد تسّنم ڕبّي ئس ئژضاڕ ئ كرا ئلّان، ها نّ ڕبّي ئسّوتل د ياد س توسّنا نّس ئ كرا ئلّان
