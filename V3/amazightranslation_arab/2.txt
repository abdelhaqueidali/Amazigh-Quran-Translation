تافوناست
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، **أليف**-**لام**-**ميم**: ڕبّي كا أد ئسّنّ ماد سرس ئواتّس؟
2 غواد أد ئݣان أدليس، ؤر ݣيس توردا، ئݣا تيفاوت ن ئمكصاض ن ڕبّي
3 غوِيلّي فلسنين س ؤنتال زون د ڕبّي، د يينيرن، د تغرغارت، أيوان أر تّژالّان، أر أكّان غ غايلّي أسن نورݣ
4 د غوِيلّي فلسنين س ما نّ فلّاك ئݣّوزن، أد ت ئݣان د **لقْࢫران** د ماد د ئݣّوزن ئزوار ك، د تمݣّاروت، سّنّ ئس د تقّان
5 غوِينّ (غْࢫين؟) مونّ د ؤغاراس ن ڕبّي نّسن، غيكلّي ݣان ويد ڕوانين
6 أمّا غوِيلّي نبلنين، ها نّ ئنݣادّا ؤكان دارسن، ئغ تن تسّيوْدت، د ئغ تن ؤر تسّيوْدت، ها نّ ؤر أد فلسن
7 أشكو ئمسل ڕبّي وولاون نّسن، د ئمزݣان نّسن، ئݣ أمدلو ف والّن نّسن، ها نّ تقّل سرسن تمّارا ئݣّوتن؟
8 ها نّ ئلّا غ ميدّن ماد ئتّينين، نفلس س ڕبّي، د واسّ أمݣّارو، ماشّ نتّني ؤر ݣين ئنافلاسن
9 أر تّكنادن ڕبّي د ويلّي فلسنين، أوا، ها نّ ؤر أد تّكنادن أ بلا ئخفاون نّسن، سول ؤر أس فريكن؟
10 أشكو تلّا تماضونت غ وولاون نّسن، س أسن ئسّوݣت ڕبّي تايّاض، د تلّا أسن تمّارا تاماݣژاژت، ف ؤيلّي سّكيركّسن
11 د ئغ أسن ئتّونّا أد ؤر تسݣوفسوم غ ييݣّي ن واكال، نّان: ها نّ أد أكّْࢫ نݣا د ئمسّاوكان
12 أوا، ها نّ نتّني س ييخفاون نّسن ݣان ئݣوفسا، ماشّ ؤر أس فريكن
13 د ئغ أسن ئتّونّا: أيوا فلسات، غيك لّي فلسن ميدّن، نّان: أوا أد نفلس غيك لّي فلسن ئمصوضن، ماشّ ؤر سرس سّينّ
