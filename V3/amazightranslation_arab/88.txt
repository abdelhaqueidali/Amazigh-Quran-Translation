88
تادولي
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئس ك ئد يوشكا ئنغميس ن تدولي؟ زعما تاغرغارت
2 ؤدماون ن ميدّن أكود أنّ نّوݣرن،  أشكو كرا سكرن يوضّڕ
3 أشكو كيݣان أد سكرن سول مرّتن ݣ واسّ ن تغرغارت
4 أر كّمضن ݣ لافييْت ئفدادّين
5 أر سّان ݣ ؤغبالو ن ؤژيژن
6 ؤر أسن ئلّي ؤمتشو بلا ݣ ؤمودروس
7 ؤر أد ئسيجّي ؤلا أر ئتّكّس لاژ
8 ئنّا ڕبّي: ماشّ ها نّ ؤدماون ياضنين ݣ واسّ ن تغرغارت فولكين فجّيجن
9 أشكو ئسّوسم تن غايلّي سكرن، ݣ تودرت
10 نتّا ڕبّي: ها نّ غوِي نّ لّان ݣ تورتيت ياتّويْن
11 ؤر أد ݣيس سّفلادن ئ ؤزرديݣ
12 ئلّا ݣيس ؤغبالو ئنݣين
13 لّانت ݣيس تيسّيوِين أتّويْنين
14 د ئزوكان تّرسنين ئ وانّا ئران أد ئسو
15 د ئسوموتن درسنين
16 د ئحنبلن فسرنين
17 نتّا ڕبّي: ماخّ ئس ؤر أد تّمنادن ئڕامان مانيك س تّوسكارن؟
18 د ئݣنّا مانيك س ياتّويْ؟
19 د ئدرارن مانيك س بيدّن؟
20 د واكال مانيك س ئتّوفسار؟
21 نتّا ڕبّي ئ ؤمازان نّس: أيوا، سّكْࢫتي ئ ميدّن، ماد أك د ئتّوزانّ،  ها نّ أد أكّْࢫ تݣيد كيّين د ؤمسّوكتي
22 ها نّ ؤر فلّاسن تݣيد أماكڕاض، ؤلا بزّيزّ
23 ماشّ وانّا ئورّين ئنبل، ئفل ليسلام
24 ها نّ راد ت ئمرّد ڕبّي تامّارا ئݣّوتن
25 ئنّا ڕبّي: ها نّ راد ؤرّين س دارنغ
26 ها نّ ئقّان د أد تن نسّحسو، كرا ئݣا ت يان د ماد ئسكر؟
