83
ئمسّفناژ
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، أنّا ئلوح ڕبّي ئمسّناژ غ ؤرواسّ
2 غوِيلّي سّيݣوتنين تاعبّارت نّغ ڕضلن غ دار ميدّن
3 ماشّ نتّني أسن ڕضلن نغ أسن ستلن أر أسن تكّيسن
4 ماخّ ئسر سّينّ غوِينّ ئس راد د سول نكرن غ ؤسمضل نّسن؟
5 غ يان واسّ ئمقّورن
6 أد ت ئݣان د واسّ لّي غ راد بيدّن مدّن ݣر ئفاسّن ن ڕبّي ن كرا ئلّان
7 عاد، ها نّ أدليس ن ئنفراغ ئلّا غ تكرژي
8 هييّا متادّ ئس تسّنت مانيك تݣا تكرژي أنّ
9 ئݣا ت ؤدليس ئتّياران
10 أرواسّ غا واسّ نّ ن ئمسكيركّسن
11 غوِيلّي سّكيركّسنين أسّ ن ؤسّحسو
12 ها نّ ؤر أد ت ئسّكيركّس بلا وانّا ئݣان أنژضام أنابكاض
13 ئغ فلّاس تّوغران أد أيلان نّغ، زعما **لقْࢫڕان** ئنّا غاياد ئݣا ؤمين ن ئمزوُورا
14 ماخّ؟ أطرس*(أطڕص) ئدل غايلّي س تّ ئنّ كّان أر ت سكارن، ؤلاون نّسن؟
15 عاد؟ ها نّ ئݣا ؤنتال ݣراسن د ڕبّي نّس أكود أنّ
16 أ وا، ها نّ راد كّمضن س تمژلا غ ؤرواسّ
17 سول ئتّونّا أسن: ها نّ غواد، أ س تّ ئنّ تكّام تسكرم ݣيس تيكركاس؟
18 عاد، ها نّ أدليس ن غوِيلّي روانين ئساتّويْ ت ڕبّي
19 هييّا متادّ ئس تسّنت مانيك تݣا تساتّويْت أنّ!؟
20 أد تّ ئݣان د ؤدليس ئتّياران
21 ئناݣان نّس ݣان تن غوِيلّي أژنين غ ئنيرن
22 ها نّ غوِيلّي روانين لّان غ تنافوت، زعما تورتيت
23 ݣونين خف ئسوما، أيوان أد تّمنادن ماد أسن ئفكا ڕبّي
24 ئغ أد ݣيسن تّمنادت أر تژڕّات غ ؤدماون نّس تاماتارت ن تنافوت د ؤفولكي نّس
25 أر سّان ئنژيڕن ئسقاقن ئحضان غ ؤفشكو نّس
26 أساݣور نّسن ئݣا توجّوت، زون د غيك أنّ أد ئݣان ما خف أد تّمحيكيرن أيت ڕبّي
27 أسركس ن ؤنژيڕ أنّ غ **تاسنيم** زعما واد ئستين باهرا
28 أد ت ئݣان د ؤغبالو لّي غ أد سّان غوِيلّي ت أژنين
29 ئنّا ڕبّي: ها نّ غوالّي ݣانين ئكنّافن، زون د **أبو جاهل** كّان تّ ئنّ أر تضصّان غ غوِيلّي فلسنين، زون د **بيلال** {أد فلّاس ئڕضو ڕبّي}
30 كّان تّ ئنّ ئغ فلّاسن زرين، أر ݣيسن تضصّان، أر سبنقاشن تيطّ؟
31 ئغ ئورّين س دار أيت دارسن، أيوان فجّيجن س ماد سكرن
32 ئغ تن ژڕان، نّان: ها نّ غوِيد ݣان ئموضّاڕ، لّيغ ضفاڕن **موحامّاد**{تاژالّيت ن ڕبّي فلّاس}
33 ئنّا ڕبّي: ها نّ ؤر د تّوزانّيكنّافن أد، ݣان ئمحضان خف ئنافلاسن؟ ماخّ أيلّيغ تن ضوفن؟
34 ئنّا ڕبّي: ها نّ غوِيلّي فلسنين غ واسّ أد، زعما أسّ ن تغرغارت أر تضصّان غ غوِيلّي نبلنين
35 سول سكّوسن خف ئسوما غ تورتيت أر تّمنادن ئمنابالن لّان غ تمّارا
36 أوا، ئس ؤمژن ئمنابالن أتيݣ ن غايلّي س تّ ئنّ كّان أر ت سكارن؟ نغ د ؤهو؟ عاد ها نّ تّياماژن
