98
ؤملان
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ؤر ݣين غوِيلّي نبلنين ݣ أيت ؤدليس، د ئمنابالن ويد نفاسّاينين أيلّيغ تن ئد يوشكا ؤملان: أد ت ئݣان د **لقوڕان** نغ د أمازان
2 نتّا أد يوزن ڕبّي ئسّاقّرا أرّاتن غوسنين، أد تن ئݣان د **لقوڕان**
3 لّان ݣيسن أيلان فاونين
4 ها نّ ؤر بضين غوِيلّي مو ئتّوفكا ؤدليس، بلا ضاڕات ن ماد أسن د يوشكا ؤملان: زعما **لقوڕان** د ؤمازان
5 ها نّ ؤر أسن ئتّونّي بلا أد سّومدن س ڕبّي، أيوان زدّيݣن أس أسݣد، مونّ د ؤغاراس ن ڕبّي أر تّژالّان، أر أكّان تاكوتي، أشكو غيك أنّ أد ئݣان أسݣد يوغدن
6 ها نّ غوِيلّي نبلنين غ أيت ؤدليس، د غوِيلّي سكرنين أمادراو ن ڕبّي، كشمن أرواسّ، زدغن بدّا ݣيس. غوِينّ أد ئݣان ؤژوض ن ؤفݣان
7 أمّا غوِيلّي فلسنين أيوان سكرن ماد ئفولكين، غوِيلّي نّ نتّنين أد ئݣان توفوت ن ؤفݣان
8 ئنّا ڕبّي: تيغراد نّسن ݣ دار ڕبّي نّسن ݣ واسّ ن تغرغارت، ݣانت تنت تورتيتين ن **عادن** لّي مو نݣين ئسافّن ݣ دّاوات نّسنت، أيوان زدغن بدّا ݣيسنت، ها نّ غوِينّ ئسدݣي فلّاسن ڕبّي، غيك لّي فلّاسن سدݣين ؤلا نتّني، ف غايلّي أسن ئفكا، ئنّا ڕبّي ها نّ غيك أنّ ئلا ئ وانّا ئكصوضن ڕبّي نّس
