1 س ييسم ن ڕبّي أمالّاي أمسمولّو،  هانّ نمّتر يان ئمّتري أد د تݣّوز تمّارا ؤر ئنتيلن
2 ف ئمنابالن، تامّارا أنّ ؤر ئلّي ماد ئژضاڕن أد تّ فلّاسن ئرار
3 تامّارا أنّ تݣّوز د غ دار ڕبّي ن ئݣنوان
4 أر دارس أقّلاين ئنيرن د ئمانّ: **جيبريل** غ واسّ تݣا تغزي نّس سمّوس ئد مراو (سمّومراو نغ د سين واݣنارن د مراو) ن ئݣيم ن ؤسݣّْࢫاس
5 أوا، أيوا ژّيضڕ، سول أژّيضڕ ئفولكين
6 هانّ نتّني غالن تامّارا ئس نّ تاݣّوݣ؟
7 ماشّ نكّني أر تّ نتّمناد تاژ د
8 غ واسّ لّي غ راد ئݣ ئݣنّا زون د أفسّاي ن ؤژڕف (نقّرت)
9 ݣين ئدرارن زون د تاضوضت
10 غيك لّي راد ئساقسا ؤمدّاكّل  غ ؤمدّاكّل نّس؟
11 واخّا تّوسّانّ ݣراتسن، سول ئݣ تايري دار ؤكنّاف أد ئسنفل ئخف نّس س وارّاو نّس، فاد أد ئنجم غ تمّارا ن واكود أنّ
12 سول د تمغارت نّس د ݣْࢫماس
13 د توْجا نّس لّيغ د ئنكر
14 د كرا ئلّان غ ئݣّي ن واكال أد هلّي ئنجم
15 أوا، هانّ ؤهو، هانّ تقّان ت ئد لعفيت
16 لّي راد ئسّكمض ئلم نّس
17 تايوا أر تاقّرا ئ والّي يورّين ياݣْࢫي **ليسلام**
18 يايوا ئسمون أيادا ئكمس ت
19 هانّ أفݣان ئتّوسكار ئݣا أمنّخشف
20 ئغ ت ئسلي ؤژڕوض ئسّغيلف
21 ئغ ت تسلي تولّوغت ئكمس
22 ماد ؤر ئݣين ئمژّالّان
23 ويلّي بدّا تّژالّان
24 د ويلّي مو تلّا تاكوتي ئتّوسّانّ غ وايادا نّسن
25 ئ ئمّتري د والّي ؤر ئطّافن
26 د ويلّي زّيغوژينّين أسّ ن ؤفرا
27 د ويلّي كصوضنين تامّارا ن ڕبّي نّسن
28 أشكو تامّارا ن ڕبّي نّسن تّوكصاض
29 د ويلّي حضانين ئكباتن نّسن
30 بلا ف تمغارين نّسن، د تّويِيوِين (تّووْيِيويِين*؟) نّسن، هانّ وينّ ؤر تّورݣامن
31 وانّا ئران ؤݣّار ن غيكانّ هانّ ئݣا غ ئنژضامن
32 د ويلّي حضانين تيغيلاف نّسن د تسوغونت نّسن
33 د ويلّي أكّانين توݣا نّسن
34 د ويلّي ف ؤر أد تزراي تژالّيت نّسن
35 وينّ: كرا ݣان ئسّمغور تن ڕبّي غ تورتيتين
36 ماخّ ماد ياغن ويلّي نبلنين أر د سرك تّاݣّْࢫان
37 أيوان ݣين تيروبّا غ تسݣا تافاسيت نّك د تژلماضت؟
38 أوا، ئس ياتّون كرا ئݣان أفݣان ݣيسن أد ئكشم تورتيت ن تنافوت؟
39 أوا، هانّ ؤهو، أشكو نسكر تن غ غايلّي نّسن؟
40 ئنّا ڕبّي: أد ݣّالغ س ڕبّي ن ؤغلّاي د ؤنتال، زعما والّي ئسكرن ئݣنوان د واكال، أ نكّني أر ئژضاڕ (نژّضاڕ*؟)
41 أد تن نسّنفل س ماد تن يوفن، هانّ ؤر ئلّي ماد ئنران ف غيكانّ
42 أجّي تن أد روّين غامان ؤر نّ تامن ديغ نماقّارن د واسّ نّسن لّي سرسن ئتّقّلن؟
43 أسّ لّي غ راد د تّفّاغن غ ئسمضال نّسن س تازّلا، زون د ئغ معابارن س تسالّافت ئبدّن؟ نغ د غيلّي ران؟
