90
تامازيرت
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي: أد ݣّالغ س تمازيرت أد، زعما **ماكّا**
2 سول كيّين ئزّري أك ڕبّي كرا تسكارد ݣ تمازيرت أد، زعما **ماكّا**
3 د ؤماراو، د كرا يورو، ئلّا ماد ئنّان **أدام** د تاروا نّس؟
4 تاݣالّيت أر ياد نسكر أفݣان ݣ تمّارا، أشكو ئمّاغ ديس ݣ تمزواروت ؤلا تامݣّاروت،
5 ئس د ئس ئغال، ؤر سار راد أس ئژضيڕ مقّار د يان؟ ها نّ ڕبّي ئژضاڕ أس
6 يايوا أر ئتّيني: سرفوفن أغ كيݣان د وايادا
7 ئس ئغال ئس ت ؤر ئژڕي يان؟ ؤلا ماد أس ئسكر؟ ها نّ ڕبّي ئژڕا ت، ئسّن ماد ئطّاف ؤلا ماد أس ئسكر
8 ئنّا ڕبّي: ماخّ ئس أس ؤر نسكير سنات والّن، أر سرسنت ئتّمناد؟
9 د ئلس أر سرس ئساول؟ د سين وانفورن ماد ئسفالكاين ؤدم نّس؟
10 غيك لّي أس نملا سين ئغاراسن: أغاراس ن ؤفولكي، د ؤغاراس ن ؤژوض،
11 ئنّا ڕبّي: ها نّ متادّ ئس كا ئمون د ؤغاراس لّي راد ت ئجّنجم؟ ݣ تمّارا ن ؤساون أنّ؟
12 هيّا متادّ ئس تسّند مانيك ئݣا ؤساون؟
13 ئنّا ڕبّي: أزراي ن ؤساون أنّ ئݣا ت، أد تسدفيد يان ئسمݣ، نغ تاوايّا
14 نغ تفكيد تيرمد ݣ يان واسّ ئلّوژن
15 ئ يان ئݣيݣيل لّي ئݣان ݣ تووْجا نّك
16 نغ أمژلوض ئݣان أجعنيض
17 سول ئݣا ݣ غوِيلّي فلسنين أيوان ݣيدّين خف ؤژّيضڕ، غيك لّي ݣادّان خف تملّاوت
18 ئنّا ڕبّي: ها نّ غوِيلّي ݣ لّانت تيميتار أد، غوِي نّ أد ئݣان أيت ئقوس
19 أمّا غوِيلّي نبلنين أيوان سّكيركّسن أيلان نّغ، ها نّ غوِينّ أد ئݣان أيت ئژّلمض، زعما أيت ؤرواسّ
20 ها نّ تورݣالن ݣ ؤݣنس ن لافييْت
