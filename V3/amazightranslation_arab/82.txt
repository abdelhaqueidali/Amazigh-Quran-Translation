82
أفروري
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي: ئغ ئقرسّي ئݣنّا
2 د ئغ لّوزّان ئتران
3 د ئغ مّركسن ئݣروالن
4 د ئغ تّوغازن ئسمضال
5 كود أنّ كرايݣات ئمان، ئسّن ماد ئزّوار، ؤلا ماد ئسّوݣّرا
6 ئنّا ڕبّي: وا أفݣان: ما ك ئسكندن خف ڕبّي نّك لّي أك ياكّان كرا تريت
7 غوا لّي ك ئسكرن ئسݣيدّي ك س ؤسقول، ئسفالكي أناو نّك
8 غيك لّي ك ئسّرواس د واناو نّا ئرا؟
9 ماخّ أيلّيغ تنّام ؤر ئلّي واسّ ن ؤسّحسو؟
10 ها نّ أر فلّاون تّاران ئنيرن لّي كن ضوفنين
11 ݣان ئزركّاتن أر تّاران، ئنّا ؤمازان: تاهالّايات غ ئنيرن أد، س ماد ئفولكين
12 ها نّ أر تّسّن كرا تسكرم
13 ئنّا ڕبّي: ها نّ غوِي لّي روانين لّان غ تنافوت
14 غيك لّي لّان غوِي لّي نبلنين غ ؤژيژن
15 أر ݣيس كّمضن غ واسّ ن ؤسّحسو
16 ها نّ نتّني ؤر سار راد د ݣيس فّوغن
17 هييّا متادّ ئس تسّنت ماد ئݣان أسّ ن ؤسّحسو: زعما تاغرغارت!؟
18 هييّا متادّ ئس تسّنت داغ ماد ئݣان أسّ ن ؤسّحسو أشكو ݣيس تامّارا!؟
19 أد ت ئݣان، د واسّ لّي غ ؤر ئسّوݣير مقّار د يان، أيما ئ وايّاض، أشكو أوال غ واسّ أنّ ئݣا وين ڕبّي
