85
ئتران
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي: أد ݣّالغ س ئݣنّا لّي غ لّان ئتران
2 د واسّ لّي س يوسيْ ڕبّي، نّان ئݣا ت واسّ ن تغرغارت؟
3 د واسّ ئنيݣي، نّان ئݣا ت واسّ ن ؤصيضياص (**لجامع**) د واسّ لّي ف ݣان ئناݣان، نّان ئݣا ت واسّ ن **عاڕافا** ماشّ ڕبّي كا أد ئسّنّ
4 ئنّا ڕبّي: ݣّولّيغ أر تّورݣامن أيت ؤسارو
5 أد ت ئݣان د لافييْت (لعفيت) لّي ݣيس ئرغان
6 أيوان سكّوسن غ تما (تاما) نّس أر ت تْتّمنادن
7 سول ݣين ئناݣان ف ماد سكارن ئ ئنافلاسن
8 ها نّ ير تن مرّتن بلا ف ئلّيغ فلسن س ڕبّي لّي ئݣان أندّابو أمتّوݣاضاي
9 لّي مو تݣا تݣلديت ن ئݣنوان د واكال تينّس، هانّ ڕبّي ئݣا أمانّاي ن كرا ئلّان
10 بانّ غوِيلّي سرفوفونّين ئنافلاسن، د تنافلاسين، غامان ؤر د ئورّين س ؤغاراس ن ڕبّي، ها نّ تلّا أسن تمّارا ن تمژلا، غيكلّي أسن تلّا تمّارا ن ؤكمّوض
11 أمّا غوِيلّي فلسنين أيوان سكرن ماد ئفولكين، هانّ لّانت أسن تورتيتين لّي مو نݣين ئسافّن غ دّاوات نّسنت، غيكانّ أد ئݣان أمركيد أنمغور
12 هانّ تايتي ن ڕبّي نّك تݣا تاسّاست ف ئمنابالن
13 هانّ نتّا أر ئسكار أفݣان ئنغي ت، يايوا ئسّودر ت ئد
14 نتّا أد ئݣان أمسّاروف أميري
15 دارس يولغن
16 أر ئسكار كرا ئرا
17 نتّا ڕبّي: ئس ك ئد يوشكا ؤنغميس ن ئزمّالن؟
18 أد ت ئݣان د **فيرعاون** د **تامود** لّي نسرفوفن
19 نغ د غوِيلّي نبلنين لّان غ توردا؟
20 أوا، هانّ ڕبّي ئسّوتل أسن د غ كرايݣات تاسݣا
21 هانّ تابرات أد تݣا **لقْࢫڕان** ئسّوسمن
22 ئتّيارا غ دار ڕبّي س غمكلّي د ديس يوشكان؟
