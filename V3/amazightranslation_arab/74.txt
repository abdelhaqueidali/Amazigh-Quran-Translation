74
أمدلاو
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي: وا أ أمدلاو، زعما غوالّي ئدلن س تملسيت، أيلّيغ د فلّاس تݣّوز تبرات ن ڕبّي، أد ت ئݣان د ؤمازان ن ڕبّي: **موحامّاد** {تاژالّيت ن ڕبّي فلّاس}
2 ئنّا أس: أيوا نيكر سّيوْد أيت **ماكّا**
3 تسّيمغورت ڕبّي نّك ف غايلّي غ أس سكرن أمادراو
4 تسغوست ؤل نّك، د تيملسيت نّك، د ݣراك د ڕبّي نّك
5 تزوݣت ف ئد ڕبّي (ڕبّين) نّسن
6 هانّ أد ؤر تاكّات، كرا ن تيكّي ف أد تامژت ؤݣّار
7 أيوا، ژيضڕ ئ ڕبّي نّك، ف غايلّي نّ فلّاك ئݣّوزن
8 هانّ ئغ ئتّوصواض غ ئسك ؤصواض ويسّ سين
9 هانّ أكود أنّ أد ئݣان أسّ أماكڕاژ (أماكراژ)
10 ف ئمنابالن ؤر ݣيس تافساسايت
11 أوا، أجّ ئيي، س غوالّي سكرغ ئݣا خيطّ، أد ت مرّتغ، ؤر ئطّاف تانّيلت
12 سول أيواغ فكيغ أس أيدا ئݣّوتن
13 د تاروا ݣانين ئناݣان
14 سول سّيݣوتغ أس باهرا غ وايادا، د تودرت، د تاروا
15 سول ئرا أد أس سّوكتغ؟
16 أوا، هانّ ؤهو، أشكو نتّا يوݣي أيلان نّغ*
17 ئنّا ڕبّي: هانّ راد ت سخنونسغ س تمّارا، أد بدّا ݣيس ئزدغ
18 هانّ ئسوِينݣم غ **لقْࢫڕان** سول ئدّس
19 أد ت ئرݣم ڕبّي مانيك س ئدّس؟
20 غيكلّي ئتّورݣام داغ غ دار ڕبّي مانيك س ئدّس؟
21 يايوا ئمناد غ ؤدماون ن أيت دارس
22 غيكلّي ئكرس ؤدم نّس: يايوا ئسكمّش ئݣنزي نّس
23 يايوا يانف ئ ؤغاراس ن ڕبّي ئسنّومغر
24 ئنّا ڕبّي: ؤر ئݣي **لقْࢫڕان** أد أ غار ئحرݣي ئتّونّان
25 د ؤر ئݣي بلا أوال ن ؤفݣان
26 ئنّا ڕبّي: هانّ غواد راد ت مضلغ غ تمژلا
27 هييّ (هييّا) متاد ئس تسّنت تامژلا أنّ!؟ هانّ توݣر كرا ئرغان
28 وانّا غ ترغا أر ت تجدّر، ؤر أد ݣيس تفّال أميا
29 أر تزلاف ئلم
30 تامژلا أنّ ضوفن تّ تژا د مراو ن وانضاف
31 هانّ ؤر نسكير غ ئمحضان ن لافييْت (لعفيت) بلا غ ئنيرن، د ؤر نسكير ؤنشك أنّ  ݣيسن أبلا أد ݣين تامروايت ئ غوِيلّي نبلنين، فاد أد سّنّ غوِيلّي مو ئتّوفكا ؤدليس ئس ئغْࢫژان ؤمازان، د أد ئدوسن غوِيلّي فلسنين ف ؤسݣد نّسن، د أد ؤر سّوردون غوِيلّي مو ؤتّوفكا ؤدليس، نتّني د ئنافلاسن، فاد أد تّينين غوِيلّي مو تلّا تماضونت غ ؤلاون نّسن، نتّني د ئمنابالن: ما س ئرا ڕبّي أد ت ئمل غ ؤمرواس أد؟ زون د غيكانّ أ س  ئتّوضّاڕ ڕبّي وانّا ئرا، غيكلّي ئسمون (ئسمونا) د ئغاراس وانّا ئرا، هانّ ؤر ئسّين ئزمّالن ن ڕبّي نّك بلا نتّا، هانّ لافييْت (لعفيت) أنّ، ؤر تݣي بلا أسّكتي ئ ميدّن
32 ئنّا ڕبّي: أد ݣّالغ س وايّور
33 د ييض ئغ ياغول
34 د تايلݣي ئغ د تغلي (تغْࢫلي)
35 أبلا أرواسّ أنّ أد ئݣان يات غ تاݣاتين مقّورنين
36 تݣا تاودا ئ ؤفݣان
37 غيكانّ ئ وانّا ݣيݣون ئران أد ئمون د ؤغاراس ن ڕبّي، نغ أس يونف،
38 كرايݣات ئمان راد ئتّيِيميژ س ماد ئسكر غ ؤرواسّ
39 بلا أيت ئفّوس، زعما، ئنافلاسن
40 هانّ نتّني لّان غ تورتيت أر تّمسقسين ݣراتسن
41 أر تّينين ئ ئكنّافن
42 ما كْࢫن (كّن) ئسّكشمن س ؤرواسّ؟
43 نّان أسن: ؤر نݣي غ ئمژّالّان؟
44 د ؤر أد ناكّا أمتشو ئ ؤمژلوض
45 غيكلّي نروّي (نرݣّْࢫي) د غوِيلّي روّينين (رݣّْࢫينين)
46 د نكّا تّ ئنّ نسّكيركّيس أسّ ن ؤفرا
47 أيلّيغ د سرنغ توشكا تمتّانت (**لموت**)
48 أكود نّا ؤر أسن تسّوݣير تالاغات ن ئمالغادن أميا
49 ماخّ ماد تن ياغن، أيلّيغ ؤنفن ئ ؤسّكتي ن ڕبّي؟
50 ئغ أسن ئتّوبدار ؤسّكتي ݣان زون د ؤدادن ئغ ديوْن
51 سول ئن روْلن (روُولن) غ ئزم
52 ئس أكّْࢫ ئرا كرايݣات أفݣان ݣيسن أد فلّاس ݣّوزن وارّاتن فسرنين؟
53 ماخّ، ئس ؤر كصوضن تامݣّاروت؟
54 عاد، هانّ **لقْࢫڕان** ئݣا أسّكتي
55 وانّا ئران أر ت ياقّرا
56 هانّ ؤر أد ت تبدرم بلا ئغ ئرا ڕبّي، نتّا أد ئسّكفان أد ئتّوكصاض، غيكلّي ياكّا أسّورف
