94
أمّوزّو
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي ئ ؤمازان نّس: ماخّ ئس أك ؤر نسمّوزي أدمر نّك ؤر ݣيس كرا ن تكرژي؟ نكتور ت س تملّاوت
2 ئنّا أس ڕبّي: نايوا نسيتّي فلّاك أبكّاض نّك، زعما ؤر سول فلّاك ئلّي مقّار د أميا، ݣ ماد ئزوارن، ؤلا ماد ئݣّْࢫران
3 لّي ئژّاين خف تاداوت نّك، زعما  ماد توسيد
4 ئنّا أس ڕبّي: نساتّويْ أك أبدار نّك، د واتيݣ نّك، كرا ݣ تّوبدارغ، ها نّ تّوبدارد ݣيس كيّين
5 ئنّا ڕبّي: ها نّ كرا ئݣا ت تيكڕژي تلّا ديس تافساست، أشكو تافساست تݣا بدّا خف تكڕژي
6 ها نّ تيكڕژي تلّا ديس تافساست غيك لّي س ئنّا ڕبّي
7 ئنّا ڕبّي: ئغ تفريد تاژالّيت نّك، نغ د تاغاوسا ن دّونيت أيوا سطّوفوڕ أس، ما راد أك ئروُو ݣراك د ڕبّي نّك
8 تسمون تن تايري نّك ݣ ؤفوس ن ڕبّي نّك
