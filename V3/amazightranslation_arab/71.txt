71
نوح
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي: هانّ نكّني نوزن نّ **نوح** س تامتّي نّس، ف أد ئسّيوْد تامتّي نّس ماحدّ ؤر تا تن ئد توشكي تمّارا تاماݣژاژت
2 ئنّا **نوح**: وا تامتّي ئنو، هانّ ݣيغ د سرون (سروْن) أمسّيوْد أمسّفرو
3 أيوا، سّومدات ئ ڕبّي، تيكصوضم ت، تسݣدم أس
4 فاد أد أون ئسّورف غ ئبكّاضن نّون، ئسّݣّرو كْࢫن أر أزمز لّي ئتّوسّانّ، هانّ أزمز ن ڕبّي ئغ د يوشكا ؤر أد سول ئتّݣرو ئغ تݣام ويد سّنّين؟
5 ئنّا ڕبّي: وا ڕبّي ئنو: هانّ نكّي غريغ ئ تمتّي ئنو غ ييض د ؤزال
6 ماشّ تيغري ئنو، ؤر أسن تسّوݣت بلا تاروُولا؟
7 هانّ نكّي أكود نّا أسن ؤكان غريغ، ف أد أسن تسّورفت، س أيوان أر تݣّان ئضوضان غ ئمزݣان نّسن، حبون ئخفاون نّسن س تملسيت نّسن، أيوان قّارن ف غيكانّ سنّومغورن باهرا؟
8 غيكلّي ؤلغ أوال ئنو أيلّيغ أسن غريغ، فاد أد سّفلدن
9 هانّ نكّني سّوݣيغ أسن د أوال ئنو ف أد أس سّفلدن، غيكلّي سرسن ساولغ أوال ئدران باهرا، ݣر ئيي ديسن
10 نّيغ أسن: أيوا، تّرات، ئ ڕبّي نّون، أد أون ئسّورف، هانّ نتّا ئݣا أمسّاروف
11 أر أون د ئتّازان أنژاڕ ئݣّوتن غ ئݣنّا
12 أر أون ياكّا أيادا، د تاروا، أر أون ئسكار تيبحارين*، غيكلّي أون ئسّنݣي ئسافّن
13 ماخّ أيلّيغ ؤر أد تسّيمغورم ڕبّي؟
14 سول ئسكر كْࢫن ياد كرا س كرا: زعما زغ تاضّيت أر أفݣان
15 ماخّ ئس ؤر أد تّمنيدم، مانيك س ئسكر سا ئݣنوان ݣان تادوليوِين؟
16 يايوا ئسكر ݣيسن أيّور أر ئسمرغا، غيكلّي ئسكر تافوكت أر تسوفو
17 هانّ ڕبّي أد كْࢫن ئسّمغين غ واكال زون د ئمغي
18 غيكلّي راد كْࢫن سرس ئرار، يايوا ئسّفغ كْࢫن ئد داغ يان ؤفوغ ياضنين ياضنين؟
19 هانّ ڕبّي ئفسر أون أكال
20 فاد أد ݣيس تّمونم د ئغاراسن لّي ݣانين ئزيلالن
21 ئنّا **نوح**: وا ڕبّي ئنو: هانّ نتّني مزارين ديتي، أيوان ضفوڕن والّي مو ؤر أد ئسّوݣت وايادا نّس د تاروا نّس أبلا أبنّان
22 هانّ سّكوندن تاكنداوت مقّورن
23 نّان: هانّ أد ؤر تبضوم د ئد ڕبّي (ڕبّين) نّون، د أد ؤر تفلم **وود** د **سووّاع** د **ياغوت** د **ياعوق** د **ناسر**
24 هانّ سوضّڕن ياد سرسن باهرا، غيكلّي ؤر أد سّيݣوتن ئ ئنژضامن أ غار تيلّاس
25 هانّ ئبكّاضن نّسن أ غ دّمن، س كشمن س لعفيت ݣّامّين كرا ن تورفيوِين ئ ئخفاون نّسن بلا ڕبّي
26 ئنّا **نوح** وا ڕبّي ئنو: هانّ ريغ ݣيك أد ؤر تاجّت مقّار د يان غ ئمنابالن، غ ئݣّي ن واكال
27 هانّ ئغ تن توجّيت، راد ؤضّڕن ئسمݣان نّك، غيكلّي ؤر راد أرون بلا أنفروغ أمنابال
28 وا ڕبّي ئنو: ريغ أد ئيي تسّورفت نكّي د ئماراون ئنو، د كرا ئكشمن س تݣمّي ئنو، ئݣ أنافلاس، غيكانّ د ئنافلاسن، د تنافلاسين، د أد ؤر تسّوݣت ئ ئنژضامن بلا تارفافانت
