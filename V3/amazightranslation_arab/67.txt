67
تاݣلديت
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئرغودا غوالّي مو تݣا تݣلديت غ ؤفوس نّس أد ت ئݣان د ڕبّي، ياتّويْ ف ئمرواسن نكرا ئسكر، نتّا أد ئژضاڕن ئ كرا ئلّان
2 نتّا أد ئسكرن تامتّانت (**لموت**) د تودرت، فاد أد كن يارم، يايوا ئستي ما مو ݣيݣون فولكين ئسكّيرن نّس، نتّا أد ئݣان أندّابو أمسّاروف
3 غوالّي ئسكرن سا ئݣنوان يان ف ئݣّي ن يان، ؤر تلّي كرا ن تمزرايت غ كرا ئسكر ؤمالّاي، أيوا رار ئژڕي س ئݣنّا ئس راد تانّيت كرا تستّيغت
4 أيوا، سيوْرّي ئژڕي سنات تيكّال، هانّ أر أك د ئتّورّاي ئژڕي ئݣا أمنّوݣر، غيكلّي ئݣا أمنّݣڕاژ
5 تاݣالّيت أر نسّوݣّلا* ياد ئݣنّا لّي د ياژن س ئتران غيكلّي ݣيسن نسكر ئمرّويْن أر كّاتن ئبلاسيوْن، نايوا نسݣودي أسن تامّارا تافدادّايت
6 غيكلّي تقّل تمّارا ن ؤرواسّ س غوِيلّي نبلنين س ڕبّي نّسن، نّخ غ تازدوغت!؟
7 ئغ نّ ݣيس ضڕن أر ݣيس سّفليدن ئ تحيطّيت نّس، يايوا أر ئتّسيس
8 ياژ د أد ئبّاقّي س تورفا، كودنّا أنّ ؤكان ݣيس ئضڕ يان ؤݣزدي، أر ت ساقسان ئمحضان نّس أر أسن تّينين، ماخّ ئس كْࢫن نّ ؤر يوشكي ؤمسّيوْد؟
9 نّان أسن عاد، يوشكا أغ ياد ؤمسّيوْد، ماشّ نسّكيركّس ت، نايوا نْنّا ؤر د ئزّوݣز ڕبّي مقّار د كرا، هانّ كنّي ؤر تلّيم بلا غ تيلّاس لّاسنين
10 نّان: متاد ئس أد نسّفليد نغ أد نسوِينݣيم ؤر راد أكّْࢫ نݣ غ أيت ؤژيژن؟
11 أيوان زّيݣزن أبكّاض نّسن، أوا أد ئمزي ڕبّي أيت تمژلا
12 هانّ غوِيلّي تّيكصوضنين ڕبّي نّسن غ ؤنتال، ئلّا أسن ؤساروف د تغراد ݣّوتنين
13 أوا، غيلاد أدرات ئ واوال نّون، نغ أس تولّم، هانّ ڕبّي ئݣا أماسّان ن كرا تسوِينݣيمّ
14 ماخّ ئس ؤر ئسّن ماد ئسكر؟ سول نتّا ئݣا أمسّولغو أموسّن
15 نتّا لّي أون ئسكر أكال ئسّانون ت، ئسّانون ت، أيوا مّودّويات س تسݣّيوِين نّس، تايوام أر تشتّام غ تيكّي نّس، هانّ ئغ د تفّاغم غ ئسمضال، دارس أد نّ ترام
16 ئس ؤر تكصوضم غوالّي س تنّام ئلّا غ ئݣنّا أد سرون ئسنّستم أكال؟ سيغ أد سروْن ئتّمدوداي؟
17 نغ د ئس ؤر تكصوضم غوالّي س تنّام ئلّا غ ئݣنّا أد كْࢫن ئسݣل س ؤمژرار (ؤمژڕاڕ)؟ ئنّا ڕبّي: هانّ راد سول تسّنم مانيك راد كْࢫن مرّتغ؟
18 تاݣالّيت ؤر أد سّكيكّسن غوِيلّي تن زوارنين، أوا مانيك سرسن تݣا تكڕژي ئنو؟
19 ماخّ ئس ؤر أد تّانّين ئݣضاض غ ئݣّي نّسن أر فسّرن ئفارّاين نّسن؟ أر تن سمونان ؤر تن يومژ غ ئݣنّا بلا أمالّاي، هانّ نتّا ئݣا أمانّاي ن كرا ئلّان
20 نغ د ماد ئݣا غواد لّي س تنّام ݣان ئنميالّالن ران أد كْࢫن ئد نّيلن بلا أمالّاي؟ هانّ ؤر لّين ئمنابالن أ غار غ تكنداوت
21 نغ د ماد ئݣا غواد لّي أون ياكّان ئغ يومژ ڕبّي أيادا نّس؟ ماشّ هانّ زّوݣازن ؤكان سول غ ؤنحيمار ديتون
22 ماخّ ئس ئنݣادّا غوالّي ئتّيدون ف ؤدم نّس ياوي تيكني، د غوالّي ئبيدّن أر ئتّمناد ئمون د ؤغاراس يوغدن؟
23 ئني: هانّ ڕبّي نتّا أد كْࢫن ئسكرن، يايوا ئسكر أون أسّفلد، د ئژڕي، د ؤلاون، ماشّ ئدروس أد ت تّالغم
24 ئني: نتّا أ كْࢫن ئݣران غ ئݣّي ن واكال، هانّ دارس أ غ راد نّ تمونم
25 أر تّينين: أوا، ماناݣو راد ييلي غيكاد ئغ تݣام ئموغْࢫژانّ؟
26 ئني: هانّ ؤر تلّي توسّنا بلا غ دار ڕبّي، هانّ أد أكّْࢫ ݣيغ نكّي د ؤمسّيوْد أمسّفرو
27 ئس ؤكان ژڕان تامّارا تاژ د، س سݣّانّ ؤدماون ن غوِيلّي نبلنين، س أسن ئتّونّا: غواد أد ئݣان غايلّي س تّ ئنّ تكّام أر ت تّغاوالم؟
28 ئني: ئس تسّنم ئغ ئيي ئسرفوفون ڕبّي نكّي ديتي ئلّان؟ نغ أغ ئسمولّا؟ ما راد ئجّنجم ئمنابالن غ تمّارا تاماݣزاژت (تاماݣژاژت)؟
29 ئني: هانّ نتّا أد ئݣان أمالّاي، نكّني نفلس سرس، نتّا أ مو نݣا تاوريت نّغ غ ؤفوس نّس، هانّ راد سول تسّنم مانوانّا ئلّان غ تيلّاس لّاسنين
30 ئني: ئس تسّنم ئغ ݣّوزن وامان نّون س تسيلا، ماتّا غوانّ ئران أد أون د ياوي أمان نݣينين؟
