تاغرغارت-2
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، 1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي: أد ݣّالغ س واسّ ن تغرغارت
2 د أد ݣّالغ س ئمان لّي تّوݣڕاژن غ ماد سكرن أ سار د  تنكرم غ ئسمضال نّون
3 ئس د ئس ئغال ؤفݣان ئس ؤر سار راد نسمون ئخسان نّس ئغ ئݣا أكال؟
4 عاد، هانّ نژضاڕ أد نسݣيدّي ئضوضان نّس
5 نغ د ئس ئرا ؤفݣان أد ئسكر غ ما ت ئد ئقّانّ تيكركاس؟
6 يايوا أر ئساقسا، ماناݣو راد ييلي واسّ ن تغرغارت
7 هانّ كود نّا ئبهض ئژڕي؟
8 سيغ ئسكّان (ئسݣّان*؟) وايّور؟
9 س تمّاقّار (تنماقّار) تافوكت د وايّور؟
10 أكود أنّ أر ئتّيني ؤفݣان ماني س راد نجمغ؟
11 ئنّا أس ڕبّي ؤهو، هانّ ؤر ئلّي ماني س راد تنجمت
12 هانّ أنموقّار غ دار ڕبّي نّك غ واسّ أنّ أد تحسوت
13 أر ئتّوغري ف ؤفݣان غ واسّ أنّ كرا ئزوار، ؤلا كرا ئسّوݣْࢫرا (ئسوݣرا)
14 هانّ أفݣان أد ئصسفاون ئخف نّس، زعما ئݣا ئنيݣي ف كرا ئسكر
15 واخّا، ئسّنتال أر ديغ ئڕمي ؤر ئسّوݣير أميا
16 ئنّا ڕبّي ئ ئمازان نّس: أد ؤر تغاوالت س ئلس نّك أحسّو ن **لقْࢫڕان** ؤرتا فلّاك ئتّوغري
17 هانّ نوسيْ أك أد ت نسمون غ ئدمارن نّك، سول نسّحسو أك ت، زاما (زعما) أد ييفسيس ف ئلس نّك
18 ئنّا أس ڕبّي، ئغ ت فلّاك نغرا، س تغري ن **جيبريل** أيوا سّفلد ئ تغري نّس
19 غيكلّي أك نوسيْ أد أك ت نسّفرو
20 نغ د ئس تميارم ختالّي د ياژن، زعما تودرت ن **دّونيت**؟
21 تايوام تفلم تامݣّاروت، ؤر أد أس تسكارم كرا؟
22 هانّ ؤدماون أكود أنّ، غ واسّ ن تغرغارت أر سوفون
23 سول أر تّمنادن غ ڕبّي نّسن، س غيكلّي د ديس يوشكان
24 غيكلّي لّان ؤدماون ياضنين أكود أنّ سݣّانّ كرسن
25 أشكو سّنّ ئس راد رفوفنّ
26 أوا، ئغ د لكمن ئمان ن ؤفݣان تاقّايت نّس
27 س ئتّونّا (ئتّياونّا) غ تاما نّس مانزا بو ئسافارن؟
28 سيغ ئسّن ؤماضون ئس د تيطّيت أيانّ؟
29 س ئمّوتّل ؤضاڕ نّس د وايّاض
30 كرا ئلّان أكود أنّ ئرا نّ دار ڕبّي نّك
31 هانّ أفݣان ؤر ئفكي تيكّي، ؤر ئژّولّا
32 ماشّ ئسّكيركّس **لقْࢫڕان** يانف ئ ؤغاراس ن ڕبّي
33 يايوا ؤمنابال أد ئدّو س دار أيت دارس أر ئتّبنغال؟
34 ئتّونّا (ئتّياونّا) أس: ها نّ توشكا د ديك تاݣات، سول تݣا أك؟
35 أشكو تݣا أك أكّْࢫ توت
36 أوا، ئس ئغال ؤفݣان أد ئغاما بلا أسقسي؟
37 ماخّ ئس تّ ئنّ ؤر ئكّي ئݣا أبرغويْ غ حلولو لّي نّ ئفّين؟
38 يايوا ئݣ تاضّيت، س ت ئسكر ڕبّي ئݣادّا
39 س ݣيس ئسكر سين واناون: أوتم د توْتمت
40 ماخّ ئس ؤر ئݣي غوالّي ئسكرن غيكانّ، واد ئژضاڕن أد ئسّودر غوِيلّي مّوتنين
