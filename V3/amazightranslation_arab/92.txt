92
ئض
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي أد ݣّالغ س ييض ئغ ئدل كرا ئلّان
2 د واسّ ئغ د يوݣّا س تيفاوت نّس
3 د غوالّي ئسكرن أوتم د تاوتمت، زون د **أدام** د **حاوّا**
4 أ تاوّوري نّون لّي تسكرم، أر تفتّس، ݣيس ماد ئفولكين، غيك لّي ݣيس ماد ؤر ئفولكين
5 ها نّ وانّا ياكّان تين ڕبّي، يايوا ئكصوض ڕبّي
6 يايوا ئفلس س ڕبّي، ئسّن ئس تلّا تورتيت ن ڕبّي
7 غوانّ راد ت نسمون د ؤغاراس ئفولكين
8 أمّا غوالّي ت يوسّن ئغال ئس ؤر ئلكيم خف ڕبّي؟
9 يايوا ئسّكيركّيس تاورّايت س دار ڕبّي
10 ها نّ راد ت نسّيكي أغاراس ن تاسّاست
11 ها نّ ؤر أد فلّاس ياسي وايادا نّس أميا ئغ ئرفوفون
12 ها نّ أساغول ئݣا وين نّغ
13 غيك لّي تݣا تمݣّاروت، ؤلا تامزواروت تين ڕبّي
14 ئنّا ڕبّي: ها نّ سّيوْدغ كْࢫن س لافييْت ئفدادّين
15 لافييْت أنّ ؤر أد سرس ئكّمض بلا واشّ
16 غوالّي ئسّكيركّسن أمازان يايوا ئنبل
17 غيك لّي راد أس يانف غوالّي ئكصوضن ڕبّي
18 غوالّي ياكّان أيادا نّس خف أد ئسغوس ئخف نّس د وايادا نّس
19 سول ؤر ئري كرا ن تنافوت ݣ دار يان س ماد ياكّا
20 بلا ؤدم ن ڕبّي لّي ياتّويْن
21 ها نّ راد سول ئلكم ماد ئرا، ݣ دار ڕبّي
