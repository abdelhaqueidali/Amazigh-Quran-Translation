63
ئزبالّاغن
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي ئ ؤمازان نّس: ئغ ك ئد ؤشكان ئزبالّاغن نّان: هانّ نݣا ئناݣان ئس تݣيت أمازان ن ڕبّي، أشكو أكوش ئسّن ئس تݣيت أمازان نّس، هانّ أكوش ئسّن ئزبالّاغن ئس ݣان ئمسّكيركّسن
2 سكرن غ تݣالّيوْت نّسن أسنتل، أيوان أنفن ئ ؤغاراس ن ڕبّي، هانّ ئنّوكما ؤيلّي س تّ ئنّ كّان أر ت سكارن
3 غيكانّ أشكو كّان تّ ئنّ فلسن، أيوان نبلن داغ، س ئمسل ڕبّي ؤلاون نّسن، هانّ نتّني ؤر أد ت سّنّ
4 ئغ أد تن تْتّمنادت أر ك تسّوسوم تيدّي نّسن، ئغ أد ساوالن أر تسّفلادت ئ واوال نّسن، زون د ئݣجوان بدّنين، أر تّغالن ئس د كرايݣاتّ تاغويّيت ئس د نتّني أ ف تلّا، نتّني ݣان أماݣزان، أيوا سضوشّي ݣيسن، أد تن ئرݣن ڕبّي، مانيك س سنفلن تيفاوت س تيلّاس؟
5 ئغ أسن ئتّونّا: أشكات ت، أد أون ئتّر ؤمازان ئ ڕبّي أساروف، أيوان سّفرغن ئخفاون نّسن، س تن تانّيت أر تّانفن ئ غيكانّ، أشكو ݣان ئمنّومغرن
6 هانّ ئݣادّا ؤكان دارسن، ئغ أسن تسّورفت، د ئغ أسن ؤر تسّورفت؟ هانّ ؤر سار راد أسن ئسّورف ڕبّي، أشكو أكوش ؤر أد ئسمون د ؤغاراس مدّن ݣانين ئنفراغ
7 نتّني أد ئتّينين، هانّ أد ؤر تاكّام ئ ويلّي لّانين غ دار ؤمازان ن ڕبّي أرد ئغ بضان، أوا هانّ لمرس* ن ئݣنوان د واكال ئݣا أ غار وين ڕبّي، ماشّ ئزبالّاغن ؤر راد ت سّنّ
8 أر تّينين أد نݣّال ئغ نورّي غر **لمادينا** أد سار ݣيس ئسّوفغ ؤمدواس أمنوݣر، هانّ تادوسي تݣا أ غار تين ڕبّي، د ؤمازان نّس، د ئنافلاسن، ماشّ ئزبالّاغن ؤر راد ت سّنّ
9 وا ويلّي فلسنين، هانّ أد كْࢫن ؤر ئسنݣيري وايادا نّون، د تاروا نّون د ؤغاراس ن ڕبّي، هانّ وانّا ئنݣاران د غيكانّ هانّ ئݣا زغ ئموكال
10 أيوا، أر أكّايات زغ أيلّي أون نفكا ماحدّ ؤرتا د توشكي **تمتّانت**س يان ݣيݣون، يايوْ أر ئتّيني: وا أكوش، متاد ئس ئيي تسوݣريت أر ئميكّ غ ؤزمز د ياژن، فاد أد أكّاغ، سول ݣغ يان زغ ئموكنان
11 هانّ أكوش ؤر سار راد ئسماتّي أزمز ن ئمان ئغ تن ئلكم ؤزمز ن **تمتّانت**، أشكو أكوش ئݣا أموسّن ن كرا تسكارم
