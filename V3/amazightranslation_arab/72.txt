72
أمدون
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي: ئتّوزان ئيي د: ئس تسّفلد يات تريبّوت غ ؤمدون، نّان: هانّ نكّني نسّفلد ئ **لقْࢫڕان** ئسّوسمن
2 أر ئمّال ماد ئفولكين، نايوا نفلس سرس، هانّ ؤر سار راد نسكر ئ ڕبّي نغ أمادراو
3 هانّ تاتّويْ تومّغري ن ڕبّي نّغ، ؤر دارس تامدّاكّلت ؤلا أرّاو
4 هانّ ئس كا تّينين ئمصوضن نّغ أزكّر ف ڕبّي
5 هانّ نكّني نغال ئس ؤر سار راد ئنّا ؤفݣان د ؤمدون تيكركاس ف ڕبّي؟
6 هانّ كّان تّ ئنّ ئرݣازن غ ؤفݣان أر تلغاسن س ئرݣازن غ ؤمدون، ئميل رنون أسن أسخنونس
7 هانّ نتّني غالن غيكلّي تغالم ئس ؤر راد د سار ئسّودر ڕبّي مقّار د يان ئغ ئمّوت؟
8 ئنّا ؤمدون: هانّ نكّني نغلي أيلّيغ ناژ ئݣنّا، ماشّ نوفا ت ئنّ ئكتار س ئمحضان دوسنين د ئمرّويْن
9 هانّ نكّني نكّا تّ ئنّ أر ݣيس نتّسكّويّوس غ مان نّ ياژن ئسّفلد، ماشّ غيلاد وانّا ئسّفليدن أر نّ ئتّافا أمرّويْ ئمدي أس
10 هانّ نكّني ؤر نسّين ئس د ؤژلوض، أ راد يوت كرا ئلّان ف ئݣّي ن واكال؟ نغ د ئس سرسن ئرا ڕبّي نّسن ماد ئفولكين؟
11 هانّ نكّني لّان ݣيݣنغ ويد روانين، غيكلّي ݣيݣنغ لّان ماد تن ؤر ئݣين، أشكو نݣا تيروبّا بضانين
12 هانّ نكّني نسّن ئس ؤر راد سار نكّ ف ڕبّي غ ئݣّي ن واكال، غيكلّي ؤر راد ت سار نسّڕمي ئغ أس نروُول
13 هانّ نكّني لّيغ نسّفلد ئ **لقْࢫڕان** نايوا نفلس سرس، أشكو وانّا ئفلسن س ڕبّي نّس، هانّ ؤر سار ئكصوض أمنّورف، ؤلا أسخنونس
14 هانّ ئلّان ݣيݣنغ ماد ئݣان **ئموسلمن** غيكلّي ݣيݣنغ ئملا ماد ئݣان ئنژضامن، غوِيلّي ݣانين **ئموسلمن** هانّ ݣْࢫران (ݣران) أفولكي
15 أمّا غوِيلّي ݣانين ئنژضامن، هانّ ݣان ئكشّوضن ن ؤرواسّ
16 هانّ متاد ئس ؤمژن أغاراس ن ڕبّي، أ سار تن نسّوا س وامان ݣّوتنين
17 فاد أد تن سرسن نارم، هانّ وانّا يوݣين أوال ن ڕبّي نّس، راد ت نسيكي ف تمّارا ئݣّوتن
18 هانّ تيمزݣيديوِين ݣانت تينّ ڕبّي، هانّ أد ݣيسنت ؤر تاقّرام ئ بلا ئ ڕبّي
19 هانّ لّيغ ئبيدّ ئسمݣ ن ڕبّي، يايوا أر أس ياقّرا، ألاين (علاين) أد فلّاس ݣودين ئمدونّ
20 ئني: هانّ أ مو أك أقّراغ د ڕبّي ئنو، هانّ ؤر أس سكيرغ أمادراو غ مقّار د يان
21  ئني: هانّ نكّي ؤر أون سّوݣيرغ أميا غ تاواغيت ؤلا تانافوت
22 ئني: هانّ ؤر سار راد ئيي ئكّس غ ؤفوس ن ڕبّي مقّار د يان، ؤلا راد سار أفغ ماني س أد تّانفغ بلا س دارس
23 بلا أسّلكم زغ دار ڕبّي د تبراتين نّس، هانّ وانّا ئمزاراين د ڕبّي د ؤمازان نّس، هانّ تقّان ت ئد لافييْت (لعفيت) ن ؤرواسّ راد بدّا ݣيس زدغن
24 أرد ئغ ژڕان غايلّي س أد تن نسّيوِيد، أكود أنّ راد سّنّ، مان وانّا مو تݣا تانّيلت نّس أجعنيض، د ما مو ئدروس ؤݣدود نّس؟
25 ئني: هانّ ؤر سّينغ ئس د ياژ غايلّي س أكْࢫن (أكن) نسّيوِيد؟ نغ د ئس راد أون ت ئنّ ئسّاݣّوݣ ڕبّي ئنو؟
26 ڕبّي أد ئݣان أماسّان ن ؤنتال، هانّ ؤر أد ئمّال أنتال نّس ئ مقّار د يان
27 بلا وانّا ف ئݣا أدݣي غ ئمازانّ، غوانّ أر ديس ييسمون (ئسمونا) ڕبّي ما تّ ئحضّون غ كرايݣات تاسݣا أرد ئسّلكم تابرات ن ڕبّي
28 فاد أد سّلكمن تيبراتين ن ڕبّي نّسن، هانّ ئسّن غيكانّ ݣيسن، أشكو ئسّوتل د ئ كرا دارسن ئلّان، غيكلّي ئسّن كرا ئلّان، يان س يان
