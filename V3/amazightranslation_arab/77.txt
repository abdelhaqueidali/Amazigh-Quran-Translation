77
تينطّفوڕين
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، أد ݣّالغ س ييسمضال ئنطّفوڕن: ئغ ئطّفاڕ كرا ݣيسن كرا
2 د تيجّاج لّي ؤسينين ئداليوْن
3 د ئسمضان لّي أطّانين ئداليوْن غ ئݣنّا
4 د ئنيرن لّي ئطّانين ماد ئرا ڕبّي ف ئمازانّ نّس
5 د غوِيلّي د زوزݣوزنين تابرات ن ڕبّي
6 تݣيتيت* أساروف، نغ تاودا
7 أ غوالّي س أون يوسيْ ڕبّي أر ئلّا، زون د أسّ ن تغرغارت، د واسّ ن ؤسّحسو
8 ئنّا ڕبّي: هانّ ئغ ݣان ئتران ئدرغان؟
9 د ئغ ئستغ ئݣنّا؟
10 د ئغ نّستمن ئدرارن؟
11 د ئغ ئسمون ڕبّي ئمازانّ غ يان واسّ؟
12 أحّ ماتّا أسّ أنّ لّي مو تن ئسّكْࢫݣرا (ئسّݣرا) ڕبّي؟
13 أد ت ئݣان د واسّ ن تيطّيت ݣر ميدّن؟
14 هييّا متاد ئس تسّنت ماد ئݣان أسّ ن تيطّيت؟
15 أرواسّ أكود أنّ ئ ئمسكيركّسن
16 ئنّا ڕبّي: ماخّ ئس ؤر نسرفوفن ئمزوُورا؟
17 نايوا نسطّفوڕ (نسضفوڕ) أسن ئمݣورا؟
18 زون د غيكانّ أد نسكار ئ ئكنّافن؟
19 أرواسّ أكود أنّ ئ ئمسكيركّسن
20 ئنّا ڕبّي: ماخّ ئس كْࢫن ؤر نسكير غ ݣار أمان؟
21 ئنّا ڕبّي: نايوا نݣين نّ أمان أنّ غ ؤزدّوغ ئتّوحضان
22 أر أزمز ئتّوسّانّ، زعما أزمز ن تلاليت
23 هانّ نسݣادّا ت، أ خّايت؟ نكّني لّي سݣادّانين؟
24 أرواسّ أكود أنّ ئ ئمسكيركّسن
25 ماخّ ئس ؤر نسكير غ واكال واد تن ئسمونّ؟
26 ئغ دّرن ؤلا ئغ مّوتن؟
27 نايوا نسكر غ ئݣّي ن واكال ئدرارن أتّويْنين، نسّو أون أمان مّيمنين
28 أرواسّ أكود أنّ ئ ئمسكيركّسن
29 أيوا، دّويات س غايلّي س تّ ئنّ تكّام تسكرم ݣيس تيكركاس، أد ت ئݣان د ؤرواسّ
30 أيوان، كشمات س لافييْت (لعفيت) ن ؤرواسّ لّي ئبضان د كڕاض ئسافّن
31 تݣا تادولي، ماشّ ؤر تݣي أسكلو، ؤما تسّنتل تامژلا
32 هانّ لافييْت (لعفيت) أر تكّات س ؤضݣيݣ زون د ئبرݣمّي
33 زون د ئڕامان سݣّانين (سݣانّين)
34 أرواسّ أكود أنّ ئ ئمسكيركّسن؟
35 غواد أد ئݣان أسّ لّي غ ؤر أد ساوالن
36 ؤلا أر أسن ئتّوسّوراي فاد أد سّورفن
37 أرواسّ أكود أنّ ئ ئمسكيركّسن
38 ئنّا ڕبّي: هانّ غواد أد ئݣان أسّ ن تيطّيت، نسمون كْࢫن، كْࢫنّي د ئمزوُورا
39 ئغ دارون، كرا ن تكنداوت؟ أيوا، كندات ئيي؟
40 أرواسّ أكود أنّ ئ ئمسكيركّسن
41 هانّ ئمكصاض ن ڕبّي، لّان غ ئسوكلا د ئغبولا
42 غيكلّي أسن لّان تݣمّوتين غ كرايݣات أناو
43 أر أسن ئتّونّا (ئتّياونّا) أيوا، شّات أر تسّام، هانّ ئرغود أون، س غايلّي س تّ ئنّ تكّام أر ت تسكارم
44 ئنّا ڕبّي: هانّ نكّْࢫني غيكانّ أسّ أد نفرّا ئ ئماواين
45 أرواسّ أكود أنّ ئ ئمسكيركّسن
46 غيكلّي ئنّا ڕبّي ئ ئمسكيركّسن: أيوا، شّات تفجّيجم ئميكّ ن ؤزمز، هانّ كْࢫنّي تݣام ئكنّافن، ترام أرواسّ
47 أرواسّ أكود أنّ ئ ئمسكيركّسن
48 ئغ أسن ئتّونّا أيوا سحونّرات ؤر أد سحونّارن: زعما ؤر أد تژالّان
49 أرواسّ أكود أنّ ئ ئمسكيركّسن
50 ماخّ مان أوال ضاڕات **لقْࢫڕان** أد س راد سول فلسن؟
