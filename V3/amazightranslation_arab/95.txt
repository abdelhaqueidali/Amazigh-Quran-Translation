95
تازارت
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبيي أد ݣّالغ س **تّين** د **زّايتون** ئلّا ماد ئنّان: تازارت د ؤزمّور لّي شتّان ميدّن؟ ئلّا ماد ئنّان: ئدرارن ݣ تمازيرت ن **لقودس**؟ زون د **طّوڕ سينين** أد ت ئݣان د ؤدرار لّي ݣ ئساول ڕبّي س **موسا** د تامازيرت ن **ماكّا** زعما ئݣّول ڕبّي س تميزار لّي ݣ د يوزن ئمازانّ نّس (ماشّ ڕبّي كا أد ئسّنّ)
2 د ؤدرار ن **سينين**
3 د تمازيرت أد: لّي ئݣدل ڕبّي ݣ كرا ئتّوكصاضن: **ماكّا** لّي ݣ د يوزن ڕبّي **موحامّاد** {تاژالّيت ن ڕبّي فلّاس}
4 ئنّا ڕبّي: تاݣالّيت أر نسكر أفݣان نساستوْ ت باهرا
5 نايوا نسّيوْرّي ت ئد س ݣار تيݣيرا، زعما أرواسّ. ئلّا ماد ئنّان أد يوسّر باهرا
6 بلا غوِيلّي فلسنين سّكرن ماد ئفولكين. غوِي نّ، لّات* أسن تغراد ؤر تبّينين ݣ دار ڕبّي
7 ئنّا ڕبّي: وا أفݣان: ما كيݣان د ؤمسّكيركّيس أيلّيغ تغالد ئس ؤر راد ييلي ؤسّحسو ݣ واسّ ن تغرغارت؟
8 ماخّ ئس ؤر ئݣي ڕبّي واد ئژضاڕن أد ئسّحسو ؤ ؤسومو؟
