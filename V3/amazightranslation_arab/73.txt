73
أمسموتّل
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي: وا أمسموتّل، زاما (زعما) غوالّي ئدلن س تملسيت نّس، أد ت ئݣان د ؤمازان ن ڕبّي
2 أيوا نيكر أر ئتّژالّا غ ييض كرا ئݣا بلا ئميكّ ݣيس
3 زون د أزݣن ن ييض، نغ ݣيس تكّيست ئميكّ؟
4 نغ أس ترنوت؟ تايوات تسّوسرت **لقْࢫڕان** أد ييمين
5 هانّ راد نّ فلّاك نزّوݣز أوال ئژّاين، أد ت ئݣان د **لقْࢫڕان**
6 أوا، هانّ تانكرا غ ييض س تژالّيت نتّات أد ئسّيلݣْࢫيغن ؤل، د أر تسفالكاي تيغري
7 هانّ أزال ئلّا أك ݣيس ؤزمز ئغزّيفن ئ تغاوسيوِين
8 أيوا، بدر ئسم ن ڕبّي نّك، تْتّوت ماد ت ؤر ئݣين
9 نتّا أد ئݣان ڕبّي ن ؤغلّاي ؤلا أنتال، ؤر ئلّي كرا ن ڕبّي بلا نتّا، أيوا ݣ تاوريت نّك غ ؤفوس ن ڕبّي
10 أيوا، ژّيضڕ ف ماد تّينين، تبضوت ديسن، تيطّيت ئفولكين
11 أوا، أجّ ئيي سرسن أد تن مرّتغ، زعما ئمسّكيركّسن غوِيلّي فجّيجنين، هانّ ژضاڕغ أسن، تسالوت أسن ئميكّ
12 هانّ لّان دارنغ ئسكرافن د لافييْت (لعفيت)، ما س أد تن نتّمرّات
13 غيكانّ د ؤمتشو ن ئكيض، د تمّارا تاماكزاژت
14 غ واسّ لّي غ أد ئتّرݣيݣي واكال د ئدرارن، غيكلّي ݣان ئدرارن أملال ئنݣين
15 هانّ نكّني، نوزن أون نّ أمازان، ئݣا فلّاون ئنيݣي زون د غيكلّي نوزن ئ **فيرعاون** أمازان
16 ماشّ ئمّزري **فيرعاون** د ؤمازان، نايوا نامژ ت يان وامّاژ ئدوسن
17 مانيك س راد تنجمّ؟ ئغ تنبلم غ يان واسّ أر تسكار تووْدا نّس غ تازّانين ئوسّارن؟
18 مقّار د ئݣنّا ئستغ س تووْدا ن واسّ أنّ؟ هانّ غايلّي س أون يّوسيْ (يوسيْ*) ڕبّي ئتّوسكار؟
19 هانّ ختيد ݣانت أسّكتي، وانّا ئران أد ئمون د ؤغاراس نّس س دار ڕبّي نّس
20 هانّ ڕبّي نّك ئسّن كيّي (كييّي) ئس أد تّيوْزت  س تژالّيت، غ سنات تيكّال شحانين ن كڕاض غ ييض د ؤزݣن نّس، د ويسّ كڕاض نّس، كيّي (كييّي) د يات تريبّوت تلّا ديك، هانّ ڕبّي أد ئسمساسان ئض (غييِيض) د واسّ، ئسّن ئس ؤر أد ت سار تسّيضنم، غيكانّ أ ف أون ئسّورف، أيوا غرات ئميكّ غ **لقْࢫڕان** أشكو ئسّن ئس راد ݣيݣون ئلين ئموضان، د وييّاض لّي تّمودّونين س تميزار، أر سيݣّيلن تولّوغت ن ڕبّي، د وييّاض أر تّمّنغان ف ؤغاراس ن ڕبّي، أوا غرات أ غار ئميكّ ݣيس، أيوا أرا تّژلّايات، أر تاكّام تاكوتي، تڕضلم ئ ڕبّي أڕطّال ئفولكين، هانّ كرا تزوارم ئ ئخفاون نّون غ ؤفولكي راد ت ئنّ تافيم غ دار ڕبّي، ئݣا أساروف، هانّ ڕبّي ئݣا أمسّاروف أمسمولّو
