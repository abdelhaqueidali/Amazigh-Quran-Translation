107
أفشكو
1 س ييسم ن ڕبّي أمالّاي أمسمولّو، ئنّا ڕبّي: ئس تسّند أ موحامّاد {تاژالّيت ن ڕبّي فلّاس} غوالّي ئسّكيركّيسن أسّ ن ؤسحسّو؟ زعما أسّ ن تغرغارت
2 أد ت ئݣان د غوالّي د ئسݣنونّين ئݣيݣيل، ئنخّل ت ف وايادا نّس
3 غيك لّي ؤر ئسّيوْس ن ؤمتشو ن ؤمژلوض
4 ئنّا ڕبّي: أرواسّ ئ ئمژّالّان ئزبالّاغن،
5 أد تن ئݣان د غوِيلّي تّونين تاژالّيت نّسن؟ ئلّا ماد ئنّان غوِيلّي تّ سݣرونين، نغ د ؤر راد ݣيس سوِينݣيمن، نغ د غوالّي سرس ؤر ئواتّسن ؤدم ن ڕبّي؟
6 غوالّي تفشارنين خف ميدّن، أد تن ژڕين؟
7 غيك لّي ؤر أر ڕطّلن أفشكو، نغ د ؤر أكّان تاكوتي
